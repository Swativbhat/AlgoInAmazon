Order Tracking

1.Breadth First Search(BFS)

BFS can be used in order tracking systems to quickly know delivery status updates across the connected delivery network.It starts from the initial delivery hub and uses a queue to visit all adjacent nodes in the delivery network. By moving through the graph level by level, it ensures that updates are spread in the shortest possible path to all connected nodes.

Code:


#include<bits/stdc++.h>
using namespace std;

// BFS for order tracking
vector<int> bfs(vector<vector<int>>& adj)  {
    int V = adj.size();
    
    int sourceHub = 0; // starting point (e.g., warehouse)
    vector<int> orderUpdateSequence;

    queue<int> q;
    vector<bool> visited(V, false);

    visited[sourceHub] = true;
    q.push(sourceHub);

    while (!q.empty()) {
        int currHub = q.front();
        q.pop();
        orderUpdateSequence.push_back(currHub);

        for (int nextHub : adj[currHub]) {
            if (!visited[nextHub]) {
                visited[nextHub] = true;
                q.push(nextHub);
            }
        }
    }

    return orderUpdateSequence;
}

int main() {
    // Node 0: Warehouse
    // Node 1: Hub A
    // Node 2: Hub B
    // Node 3: Delivery Center
    // Node 4: Final Destination
    vector<vector<int>> adj = {
        {1, 2}, 
        {0, 2, 3}, 
        {0, 4}, 
        {1, 4}, 
        {2, 3}
    };

    vector<int> trackingUpdates = bfs(adj);

    cout << "Order Status Update Path: ";
    for (auto hub : trackingUpdates) {
        cout << hub << " ";
    }
    cout << endl;

    return 0;
}

Efficiency Analysis

Time Complexity Space Complexity
O(V + E) O(V)

Note

  • V is the number of vertices,
  • E is the number of edges in the graph
